CREATE TABLE
  `products_features` (
    `id` int NOT NULL AUTO_INCREMENT,
    `product` int NOT NULL,
    `feature` int NOT NULL,
    PRIMARY KEY (`id`),
    KEY `products_features_relation_feature` (`feature`),
    KEY `products_features_relation_product` (`product`),
    CONSTRAINT `products_features_relation_features` FOREIGN KEY (`feature`) REFERENCES `features` (`id`),
    CONSTRAINT `products_features_relation_product` FOREIGN KEY (`product`) REFERENCES `products` (`id`)
  ) ENGINE = InnoDB DEFAULT CHARSET = utf8mb4 COLLATE = utf8mb4_0900_ai_ci
