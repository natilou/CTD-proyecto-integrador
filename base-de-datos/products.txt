CREATE TABLE
  `products` (
    `id` int NOT NULL AUTO_INCREMENT,
    `title` varchar(45) NOT NULL,
    `category` int NOT NULL,
    `address` varchar(100) NOT NULL,
    `city` int NOT NULL,
    `description` varchar(255) DEFAULT NULL,
    PRIMARY KEY (`id`),
    UNIQUE KEY `id_UNIQUE` (`id`),
    KEY `products_relation_category` (`category`),
    KEY `products_relation_city` (`city`),
    CONSTRAINT `products_relation_category` FOREIGN KEY (`category`) REFERENCES `categories` (`id`),
    CONSTRAINT `products_relation_city` FOREIGN KEY (`city`) REFERENCES `cities` (`id`)
  ) ENGINE = InnoDB AUTO_INCREMENT = 4 DEFAULT CHARSET = utf8mb4 COLLATE = utf8mb4_0900_ai_ci
