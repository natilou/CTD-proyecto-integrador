CREATE TABLE
  `cities` (
    `id` int NOT NULL,
    `name` varchar(45) NOT NULL,
    `country` int NOT NULL,
    PRIMARY KEY (`id`),
    UNIQUE KEY `id_UNIQUE` (`id`),
    KEY `fk_cities_country` (`country`),
    CONSTRAINT `fk_cities_country` FOREIGN KEY (`country`) REFERENCES `countries` (`id`)
  ) ENGINE = InnoDB DEFAULT CHARSET = utf8mb4 COLLATE = utf8mb4_0900_ai_ci
